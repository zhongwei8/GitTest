```git
$ git init                  # 初始化本地分支
$ git branch -M main        # 重命名本地分支
$ git remote add origin git@github.com:zhongwei8/GitTest.git    # 添加远程分支
$ git push -u origin main   # 推送到远端


$ git add git操作.md            # 添加文件
$ git commit -m '添加git操作'    # 添加注释
$ git push -u origin main       # 同步分支

```
